压缩网络图片并下载

# 依赖项

  • Hutool: 一个小而全的Java工具类库。
  1. 添加 Hutool 依赖
<dependency>
            <groupId>cn.hutool</groupId>
            <artifactId>hutool-all</artifactId>
            <version>5.5.3</version>
        </dependency>
  1. Service
 @Service
public class Download {

    public File[]  getZip() {
        File[] arrays = new File[2];
        File file1 = UrlToFile("http://img7.dl.ltimg.net/emoji/egg/5b5a8fd525495912c2ed7068?imageView2/1/w/300/h/300");
        File file2 = UrlToFile("http://img7.dl.ltimg.net/emoji/egg/5b5a8fd525495912c2ed7068?imageView2/1/w/300/h/300");
        arrays[0] = file1;
        arrays[1] = file2;
        return arrays;
    }


    public File UrlToFile(String imageUrl) {

        URL url;
        HttpURLConnection urlConnection = null;
        InputStream inputStream = null;
        ByteArrayOutputStream baos = null;
        File file = null;
        FileOutputStream fout = null;
        try {
            url = new URL(imageUrl);
            urlConnection = (HttpURLConnection) url.openConnection();
            urlConnection.connect();
            inputStream = urlConnection.getInputStream();

            baos = new ByteArrayOutputStream();

            byte[] buffer = new byte[1024];
            int len = 0;
            while ((len = inputStream.read(buffer)) != -1) {
                baos.write(buffer, 0, len);
            }
            file = File.createTempFile(IdUtil.simpleUUID(), ".jpg");
            fout = new FileOutputStream(file);
            baos.writeTo(fout);

        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            if (fout != null) {
                try {
                    fout.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
            if (inputStream != null) {
                try {
                    inputStream.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
            if (baos != null) {
                try {
                    baos.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
            if (urlConnection != null) {
                urlConnection.disconnect();
            }
        }

        return file;

    }


}

    @RequestMapping(value = "/download", method = RequestMethod.GET)
    public void downloadCustomerExcel(HttpServletResponse response) throws UnsupportedEncodingException {

        File [] array = download.getZip();
        response.setContentType("application/octet-stream");
        response.setHeader("Content-Disposition", "attachment;filename="+new String("压缩包".getBytes("UTF-8"), "ISO-8859-1")+".zip");
        try ( OutputStream outputStream =  response.getOutputStream()){
            ZipUtil.zip(outputStream, CharsetUtil.defaultCharset(),false, null, array);
            outputStream.flush();
        } catch (IOException e) {
            e.printStackTrace();
            System.out.println("出大事了， 系统出现异常了！");
        }

    }

步骤

  • URL：统一资源定位符，建立连接。
  • 读取字节生成文件，多文件封装成数组。
  • 使用ZipUtil 压缩并且写入输出流。
